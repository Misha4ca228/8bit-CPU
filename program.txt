# --- Stack test: PUSH A, PUSH B, POP -> should output "BA" ---

    LDI 0, 17;        # R0 = 'H'? нет, у тебя H=17. Нам нужны A=10, B=11
    LDI 1, 10;        # R1 = 'A'
    LDI 2, 11;        # R2 = 'B'

    PUSH 1;           # push 'A'
    PUSH 2;           # push 'B'

    POP 3;            # R3 = 'B'
    POP 4;            # R4 = 'A'

    # Пишем в видеобуфер (0xFFF0 = 65520)
    ST 3, 65520;      # screen[0] = 'B'
    ST 4, 65521;      # screen[1] = 'A'

    HALT;
